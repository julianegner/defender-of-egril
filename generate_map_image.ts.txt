/*
 * Egril Map Image Generator
 * =========================
 * Generates terrain map images for Defender of Egril from JSON map files.
 *
 * Rendering approach ported from mapgen4 (https://github.com/redblobgames/mapgen4)
 * Copyright 2018-2025 Red Blob Games <redblobgames@gmail.com>, Apache-2.0
 *
 * Pipeline (identical in spirit to mapgen4's render.ts / map.ts):
 *
 *  1. Tile biome params  Each tile type gets (elevation, moisture, noiseAmp).
 *                        noiseAmp controls how much procedural variation is
 *                        added: high for NO_PLAY (jagged mountains), near-zero
 *                        for flat BUILD_AREA / PATH tiles.
 *
 *  2. Gaussian blend     For every pixel, Gaussian-weight nearby hex centres
 *                        to get a smooth base elevation and moisture (handles
 *                        biome-boundary transitions without hard hex edges).
 *
 *  3. Simplex noise      Multi-octave 2-D simplex noise (same library as
 *                        mapgen4) is added to elevation, scaled by noiseAmp.
 *                        Low-moisture mountains become jagged; flat areas stay
 *                        flat.  Seed is derived from the map id → deterministic.
 *
 *  4. Hillshading        Surface normals are computed from the elevation
 *                        gradient (central differences).  A directional light
 *                        (northwest, 45° elevation) is dotted against the
 *                        normal → bright faces / shadow faces, exactly as
 *                        mapgen4's frag_drape shader does.
 *
 *  5. Outline            Steep slope regions (high gradient magnitude, like
 *                        mapgen4's u_outline_strength logic) are darkened to
 *                        produce the characteristic black mountain-ridge
 *                        outlines.
 *
 *  6. Colormap           mapgen4's biome colormap (colormap.ts) maps the final
 *                        (elevation, moisture) to RGB; shading and outline
 *                        factors are applied to get the final pixel colour.
 *
 * Biome mapping:
 *   NO_PLAY     → mountains  (elevation=0.60, moisture=0.00, noiseAmp=0.45)
 *   RIVER       → water      (elevation=-0.30, moisture=0.00, noiseAmp=0.05)
 *   PATH        → arid/sandy (elevation=0.06, moisture=0.08, noiseAmp=0.04)
 *   BUILD_AREA  → grassland  (elevation=0.10, moisture=0.85, noiseAmp=0.04)
 *   SPAWN_POINT → same as PATH
 *   TARGET      → same as PATH
 *
 * Usage:
 *   node dist/generate_map_image.js maps/defender-of-egril/map_tutorial.json
 *   node dist/generate_map_image.js maps/defender-of-egril/map_tutorial.json out.png
 *   node dist/generate_map_image.js --all
 *   node dist/generate_map_image.js --all --output-dir my_output/
 */

import * as fs from 'node:fs';
import * as path from 'node:path';
import { PNG } from 'pngjs';
import { createNoise2D } from 'simplex-noise';
import { sampleColormap } from './colormap.js';
import { hexCenter, imageSize, HEX_WIDTH, BLEND_SIGMA } from './hexgrid.js';

// ---------------------------------------------------------------------------
// Tile type → biome parameters
// ---------------------------------------------------------------------------

interface Biome {
    /** Base elevation fed into the mapgen4 colormap. */
    elevation: number;
    /** Moisture fed into the mapgen4 colormap. */
    moisture: number;
    /** Amplitude of simplex noise added to elevation.
     *  High → jagged mountains; near-zero → flat terrain. */
    noiseAmp: number;
}

const TILE_BIOME: Record<string, Biome> = {
    NO_PLAY:     { elevation:  0.60, moisture: 0.00, noiseAmp: 0.45 },
    RIVER:       { elevation: -0.30, moisture: 0.00, noiseAmp: 0.05 },
    PATH:        { elevation:  0.06, moisture: 0.08, noiseAmp: 0.04 },
    // Maximum moisture (1.0) ensures clearly visible green even after Gaussian
    // blending dilutes it with adjacent low-moisture PATH tiles.
    BUILD_AREA:  { elevation:  0.05, moisture: 1.00, noiseAmp: 0.04 },
    SPAWN_POINT: { elevation:  0.06, moisture: 0.08, noiseAmp: 0.04 },
    TARGET:      { elevation:  0.06, moisture: 0.08, noiseAmp: 0.04 },
};
const DEFAULT_BIOME: Biome = TILE_BIOME.NO_PLAY;

// ---------------------------------------------------------------------------
// Rendering constants (tuned to match mapgen4's visual style)
// ---------------------------------------------------------------------------

/** Scales the elevation gradient when computing the surface normal.
 *  Higher → steeper apparent slopes → stronger 3-D shading effect.
 *  Increased proportionally with NOISE_BASE_PX: larger base frequency →
 *  shallower per-pixel gradients → higher scale needed to maintain strong
 *  shadows. */
const SHADING_SCALE = 160.0;
/** Minimum light level (0-1); lower values give darker shadow faces.
 *  Set to 0.22 (slightly higher than the mapgen4 minimum) so that shadow
 *  faces are clearly dark but not so close to black that they appear as
 *  solid dark bands / "strange lines" along hex-row boundaries. */
const AMBIENT       = 0.22;
/** Light direction: northwest (315°), 45° elevation, in image coordinates
 *  (x right, y down).  Vector points FROM surface TOWARD the light source. */
const LIGHT = Object.freeze([-0.5, -0.5, 0.707] as const); // pre-normalised (len≈1)

/** Multiplier for the slope-magnitude outline effect (mapgen4: u_outline_strength). */
const OUTLINE_STRENGTH  = 15.0;
/** Threshold below which no outline darkening is applied. Relative to
 *  OUTLINE_STRENGTH so the effective onset slopeDiff ≈ OUTLINE_THRESHOLD/OUTLINE_STRENGTH. */
const OUTLINE_THRESHOLD = 0.25;

/** How strongly high-moisture (BUILD_AREA) tiles spread their colour into
 *  neighbours during the Gaussian blend.  Each tile's Gaussian weight is
 *  multiplied by (1 + moisture * MOISTURE_DOMINANCE), so BUILD_AREA tiles
 *  (moisture=1.0) contribute 1+MOISTURE_DOMINANCE× as much as zero-moisture
 *  tiles at the same distance.  This makes green areas bleed into PATH/NO_PLAY
 *  boundary zones instead of the other way around. */
const MOISTURE_DOMINANCE = 3.0;

// Multi-octave simplex noise: 3 octaves, each at double the previous frequency.
//   Octave 1 (low freq)  → broad, rounded mountain shapes (~1–2 hexes across)
//   Octave 2 (mid freq)  → secondary ridges and valleys
//   Octave 3 (high freq) → fine surface texture
const NOISE_SCALES  = [1.0, 2.0, 4.0] as const;
// High weight on the first octave → smoother, more distinct primary peaks
// (closer to the mapgen4 look of clear individual mountain bumps).
// Weights sum to 1.0 (verified: 0.70+0.22+0.08 = 1.00).
const NOISE_WEIGHTS = [0.70, 0.22, 0.08] as const;
/** Base spatial frequency for mountain noise.
 *  1 peak spans approximately NOISE_BASE_PX pixels.
 *  Larger value → fewer, broader mountain folds (closer to the mapgen4
 *  reference where each peak occupies ~3 hex widths).
 *  HEX_WIDTH * 3 gives ~1 distinct peak per 3-hex block, matching the
 *  reference image of large rounded mountains with clear valleys between. */
const NOISE_BASE_PX = HEX_WIDTH * 3;

// ---------------------------------------------------------------------------
// Spatial grid for O(1) per-pixel hex-lookup
// ---------------------------------------------------------------------------

class SpatialGrid {
    private readonly cells: Map<number, number[]> = new Map();
    private readonly gridW: number;

    constructor(private readonly cellSize: number, imageWidth: number) {
        this.gridW = Math.ceil(imageWidth / cellSize) + 2;
    }

    private key(cx: number, cy: number): number {
        return cy * this.gridW + cx;
    }

    insert(x: number, y: number, idx: number): void {
        const cx = Math.floor(x / this.cellSize);
        const cy = Math.floor(y / this.cellSize);
        const k = this.key(cx, cy);
        if (!this.cells.has(k)) this.cells.set(k, []);
        this.cells.get(k)!.push(idx);
    }

    query(px: number, py: number, r: number): number[] {
        const result: number[] = [];
        const cx = Math.floor(px / this.cellSize);
        const cy = Math.floor(py / this.cellSize);
        for (let dx = -r; dx <= r; dx++) {
            for (let dy = -r; dy <= r; dy++) {
                const cell = this.cells.get(this.key(cx + dx, cy + dy));
                if (cell) for (const i of cell) result.push(i);
            }
        }
        return result;
    }
}

// ---------------------------------------------------------------------------
// Map data types
// ---------------------------------------------------------------------------

interface MapData {
    id: string;
    name: string;
    width: number;
    height: number;
    tiles: Record<string, string>;
    riverTiles?: Record<string, unknown>;
}

function loadMapData(filePath: string): MapData {
    const raw = JSON.parse(fs.readFileSync(filePath, 'utf-8'));
    return (raw.data ?? raw) as MapData;
}

// ---------------------------------------------------------------------------
// Deterministic seeded PRNG (used to seed simplex-noise)
// ---------------------------------------------------------------------------

function hashString(str: string): number {
    let h = 0x811C9DC5 >>> 0;
    for (let i = 0; i < str.length; i++) {
        h = Math.imul(h ^ str.charCodeAt(i), 0x01000193) >>> 0;
    }
    return h;
}

function makeRandFloat(seed: number): () => number {
    let s = (seed >>> 0) || 1;
    return (): number => {
        s = (Math.imul(1664525, s) + 1013904223) >>> 0;
        return s / 0x100000000;
    };
}

// ---------------------------------------------------------------------------
// Core image generation
// ---------------------------------------------------------------------------

export function generateMapImage(mapFile: string, outputFile?: string): string {
    const mapData = loadMapData(mapFile);
    const gridW = mapData.width;
    const gridH = mapData.height;
    const tiles = mapData.tiles ?? {};

    function tileAt(gx: number, gy: number): string {
        return tiles[`${gx},${gy}`] ?? 'NO_PLAY';
    }

    const [imgW, imgH] = imageSize(gridW, gridH);

    // ── Build hex-centre arrays ────────────────────────────────────────────
    const n = gridW * gridH;
    const cX        = new Float32Array(n);
    const cY        = new Float32Array(n);
    const elevBase  = new Float32Array(n);
    const moistBase = new Float32Array(n);
    const noiseAmps = new Float32Array(n);

    const sigma      = BLEND_SIGMA;
    const sigma2     = 2 * sigma * sigma;
    const cellSize   = sigma;
    // Search ±4 cells so the nearest undetected hex is ≥4σ away, where the
    // Gaussian weight is exp(−8) ≈ 0.03% — well below visibility.
    // Using ±3 cells would leave a hex at ~3.1σ undetected from one side of
    // a cell boundary but detected from the other, producing hairline dark or
    // light lines every ~41 px (one cell width) across the image.
    const searchR    = Math.ceil(4 * sigma / cellSize);

    const grid = new SpatialGrid(cellSize, imgW);

    let idx = 0;
    for (let gx = 0; gx < gridW; gx++) {
        for (let gy = 0; gy < gridH; gy++) {
            const [cx, cy] = hexCenter(gx, gy);
            cX[idx] = cx;
            cY[idx] = cy;
            const b = TILE_BIOME[tileAt(gx, gy)] ?? DEFAULT_BIOME;
            elevBase [idx] = b.elevation;
            moistBase[idx] = b.moisture;
            noiseAmps[idx] = b.noiseAmp;
            grid.insert(cx, cy, idx);
            idx++;
        }
    }

    // Seeded noise keyed to the map id for deterministic, reproducible output.
    // mapData.id is required by the map format; fallback to a fixed string so
    // novel/hand-crafted maps still produce consistent images.
    const noiseId = mapData.id ?? 'egril-default';
    const noise2D = createNoise2D(makeRandFloat(hashString(noiseId)));

    // ── Pass 1 · elevation + moisture per pixel ────────────────────────────
    // Stores intermediate values needed for hillshading (pass 2).
    const elevMap  = new Float32Array(imgW * imgH);
    const moistMap = new Float32Array(imgW * imgH);
    // Per-pixel blended noiseAmp, stored so pass 2 can mask the outline effect
    // to mountain regions only (prevents dark contour lines at flat biome
    // boundaries caused by the Gaussian elevation gradient).
    const naMap    = new Float32Array(imgW * imgH);

    for (let py = 0; py < imgH; py++) {
        for (let px = 0; px < imgW; px++) {
            const nearby = grid.query(px, py, searchR);
            let tW = 0, aE = 0, aM = 0, aNA = 0;
            for (const hi of nearby) {
                const dx = px - cX[hi], dy = py - cY[hi];
                // High-moisture tiles (BUILD_AREA) get extra Gaussian weight so
                // they dominate transition zones → green spreads INTO path/mountain
                // rather than green being diluted by its neighbours.
                const moistureW = 1.0 + moistBase[hi] * MOISTURE_DOMINANCE;
                const w = moistureW * Math.exp(-(dx * dx + dy * dy) / sigma2);
                aE  += w * elevBase [hi];
                aM  += w * moistBase[hi];
                aNA += w * noiseAmps[hi];
                tW  += w;
            }

            let e = aE / tW;
            const m  = aM  / tW;
            const na = aNA / tW;

            // Multi-octave simplex noise: amplitude proportional to biome noiseAmp.
            // mapgen4 uses the same approach: hills/mountains get high amplitude,
            // flat land gets near-zero amplitude.
            if (na > 0.005) {
                let noiseVal = 0;
                for (let oct = 0; oct < 3; oct++) {
                    const f = NOISE_SCALES[oct] / NOISE_BASE_PX;
                    noiseVal += NOISE_WEIGHTS[oct] * noise2D(px * f, py * f);
                }
                // Clamp to [-1, 1]: the mapgen4 colormap is defined over this
                // range (negative = ocean, positive = land up to snowy peaks).
                e = Math.max(-1, Math.min(1, e + na * noiseVal));
            }

            elevMap [py * imgW + px] = e;
            moistMap[py * imgW + px] = m;
            naMap   [py * imgW + px] = na;
        }
    }

    // ── Pass 2 · hillshading + outlines + colormap ─────────────────────────
    // Mirrors mapgen4's frag_drape shader logic.
    const png = new PNG({ width: imgW, height: imgH });
    const [lx, ly, lz] = LIGHT;

    for (let py = 0; py < imgH; py++) {
        for (let px = 0; px < imgW; px++) {
            const i = py * imgW + px;
            const e = elevMap[i];
            const m = moistMap[i];
            const na = naMap[i];

            // Neighbour indices (clamped at image edges)
            const iL = i - (px > 0        ? 1     : 0);
            const iR = i + (px < imgW - 1 ? 1     : 0);
            const iU = i - (py > 0        ? imgW  : 0);
            const iD = i + (py < imgH - 1 ? imgW  : 0);

            const eL = elevMap[iL], eR = elevMap[iR];
            const eU = elevMap[iU], eD = elevMap[iD];

            // Surface normal from elevation gradient (central differences),
            // scaled to amplify slope angles.  Normal = (-gx, -gy, 1)/|...|
            const gx = (eR - eL) * 0.5 * SHADING_SCALE;
            const gy = (eD - eU) * 0.5 * SHADING_SCALE;
            const nLen = Math.sqrt(gx * gx + gy * gy + 1);
            const nx = -gx / nLen, ny = -gy / nLen, nz = 1 / nLen;

            // Lambertian shading: ambient + max(0, N · L).
            // Capped at 2.0 (not 1.0) to allow over-bright specular highlights
            // on lit mountain peaks, matching mapgen4's visual style.
            const shade = Math.min(2.0,
                AMBIENT + Math.max(0, nx * lx + ny * ly + nz * lz));

            // Outline: darken steep-slope pixels (mapgen4 u_outline_strength).
            // Squared formula concentrates the effect at the steepest ridgelines
            // only, producing thin sharp dark contours around each peak.
            //
            // The outline mask (noiseAmp / 0.25, clamped 0-1) restricts the
            // effect to mountain regions (NO_PLAY noiseAmp ≈ 0.45) and fades it
            // out over flat biomes (PATH/BUILD_AREA noiseAmp ≈ 0.04).  Without
            // this mask the Gaussian elevation gradient at biome boundaries is
            // strong enough to trigger the outline formula, creating unwanted
            // dark straight lines at hex-row transitions.
            const slopeDiff = Math.abs(eR - eL) + Math.abs(eD - eU);
            const outlineRaw = Math.max(0, slopeDiff * OUTLINE_STRENGTH - OUTLINE_THRESHOLD);
            const outlineMask = Math.min(1, na / 0.25);
            const outline    = Math.min(1, outlineRaw * outlineRaw) * outlineMask;

            const [r, g, b] = sampleColormap(e, m);
            const s = shade * (1 - outline);

            const p = i * 4;
            png.data[p]     = Math.round(Math.min(255, Math.max(0, r * s)));
            png.data[p + 1] = Math.round(Math.min(255, Math.max(0, g * s)));
            png.data[p + 2] = Math.round(Math.min(255, Math.max(0, b * s)));
            png.data[p + 3] = 255;
        }
    }

    const dest = outputFile ?? mapFile.replace(/\.json$/i, '.png');
    fs.mkdirSync(path.dirname(dest), { recursive: true });
    // Use UP row-filter (filterType=2) + Z_DEFAULT_STRATEGY for best compression
    // on terrain images with Gaussian-blended gradients.  Reduces file size ~34%
    // compared to pngjs defaults with no change to pixel dimensions or quality
    // (PNG is lossless; filters only affect compression, not the decoded image).
    fs.writeFileSync(dest, PNG.sync.write(png, { deflateLevel: 9, deflateStrategy: 0, filterType: 2 }));
    console.log(`  Generated: ${dest}`);
    return dest;
}

// ---------------------------------------------------------------------------
// CLI
// ---------------------------------------------------------------------------

const DEFAULT_MAPS_DIR = path.join(process.cwd(), 'maps', 'defender-of-egril');
const DEFAULT_OUT_DIR  = path.join(process.cwd(), 'output', 'defender-of-egril');

function main(): void {
    const args = process.argv.slice(2);
    const allMode = args.includes('--all');

    // Parse --output-dir value
    let outDir = DEFAULT_OUT_DIR;
    const positional: string[] = [];
    for (let i = 0; i < args.length; i++) {
        if (args[i] === '--output-dir') {
            outDir = args[++i] ?? DEFAULT_OUT_DIR;
        } else if (args[i] === '--all') {
            // handled above
        } else if (!args[i].startsWith('--')) {
            positional.push(args[i]);
        }
    }

    if (allMode || positional.length === 0) {
        // Batch mode: process all .json files in the default maps directory
        const mapFiles = fs.readdirSync(DEFAULT_MAPS_DIR)
            .filter(f => f.endsWith('.json'))
            .sort()
            .map(f => path.join(DEFAULT_MAPS_DIR, f));

        if (mapFiles.length === 0) {
            console.error(`No .json map files found in ${DEFAULT_MAPS_DIR}`);
            process.exit(1);
        }

        fs.mkdirSync(outDir, { recursive: true });
        console.log(`Processing ${mapFiles.length} map(s) → ${outDir}`);

        let errors = 0;
        for (const mf of mapFiles) {
            try {
                generateMapImage(mf, path.join(outDir, path.basename(mf).replace('.json', '.png')));
            } catch (err) {
                console.error(`  ERROR ${path.basename(mf)}: ${err}`);
                errors++;
            }
        }

        if (errors > 0) {
            console.error(`\n${errors} map(s) failed.`);
            process.exit(1);
        }
        console.log(`\nDone. ${mapFiles.length} image(s) written to ${outDir}`);
    } else {
        // Single map mode
        const mapFile  = positional[0];
        const outFile  = positional[1];

        if (!mapFile) {
            console.error('Usage: generate_map_image <map_file.json> [output.png]');
            console.error('       generate_map_image --all [--output-dir <dir>]');
            process.exit(1);
        }

        try {
            generateMapImage(mapFile, outFile);
        } catch (err) {
            console.error(`Error: ${err}`);
            process.exit(1);
        }
    }
}

main();
